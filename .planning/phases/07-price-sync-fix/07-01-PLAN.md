    **Root cause:** `update_sale_price()` (line 2476) trusts the API response code without verifying the price actually changed. The Coupang API returns `{"code": "SUCCESS"}` but the price is NOT applied on Wing.

    **Fix — add read-back verification:**

    1. In `update_sale_price()` (line 2476), after getting SUCCESS response:
       - Log the FULL response body: `_log_sync.info(f"판매가 변경 API 응답: vid={vendor_item_id} price={new_price} response={result}")`
       - Wait 1 second for Coupang propagation: `await asyncio.sleep(1.0)`
       - Call `get_vendor_item_stock(vendor_item_id)` to read back current price
       - Extract `salePrice` from response (same logic as `_get_current_sale_price`)
       - If read-back price == new_price: verified success, return True
       - If read-back price != new_price: retry once more after 2 seconds
       - If still mismatched after retry: log error with details (expected vs actual) and return False

    2. Exact code structure:
       ```python
       async def update_sale_price(vendor_item_id: str, new_price: int) -> bool:
           if new_price % 10 != 0:
               _log_sync.error(f"판매가 변경 실패 → 10원 단위만 허용 ...")
               return False

           path = f"{COUPANG_SELLER_MARKETPLACE}/vendor-items/{vendor_item_id}/prices/{new_price}"
           try:
               result = await _coupang_put(path, params={"forceSalePriceUpdate": "true"})
               code = str(result.get("code", ""))
               _log_sync.info(
                   f"판매가 변경 API 응답: vid={vendor_item_id} "
                   f"price={new_price:,} response={result}"
               )
               if code not in ("SUCCESS", "200"):
                   _log_sync.error(f"판매가 변경 실패 → {result}")
                   return False
           except Exception as e:
               _log_sync.error(f"판매가 변경 예외: {e}")
               return False

           # Read-back verification
           for attempt in range(2):
               await asyncio.sleep(1.5 if attempt == 0 else 3.0)
               try:
                   api_data = await get_vendor_item_stock(vendor_item_id)
                   actual = None
                   if api_data:
                       for key in ("salePrice", "price"):
                           val = api_data.get(key)
                           if val is not None:
                               try:
                                   actual = int(val)
                               except (ValueError, TypeError):
                                   pass
                               if actual and actual > 0:
                                   break
                               actual = None
                   if actual == new_price:
                       _log_sync.info(
                           f"판매가 변경 검증 성공: vid={vendor_item_id} "
                           f"price={new_price:,}원 (attempt {attempt+1})"
                       )
                       return True
                   _log_sync.warning(
                       f"판매가 변경 검증 불일치 (attempt {attempt+1}/2): "
                       f"vid={vendor_item_id} expected={new_price:,} actual={actual}"
                   )
               except Exception as e:
                   _log_sync.warning(f"판매가 검증 조회 실패: {e}")

           _log_sync.error(
               f"판매가 변경 검증 최종 실패: vid={vendor_item_id} "
               f"expected={new_price:,} — API said SUCCESS but price not applied"
           )
           return False
       ```
